.addClass(className)

Adds the specified class to each element in the set of matched elements.

$("p:nth-of-type(2)").addClass("adding_class");

.removeClass(className)

Removes a single, multiple, or all classes from each element in the set of matched elements.

$("p:nth-of-type(2)").removeClass("adding_class");

.after(content)

Insert content, specified by the parameter, after each element in the set of matched elements.

$("#this_p").after("<span class='red'> Added after first paragraph</span>");

.append(content[, content])

Insert content, specified by the parameter, to the end of each element in the set of matched elements.

$("#this_p2").append("<span class='red'> Appended to this paragraph</span>");

.attr(attributeName)

Get the value of an attribute for the first element in the set of matched elements or set one or more attributes for every matched element.

$("#this_p3").text($("img").attr("alt"));

.before(content)

Insert content, specified by the parameter, before each element in the set of matched elements.

$("#this_p4").before("<span class='red'>Added before paragraph</span>");

.html()

Get the HTML contents of the first element in the set of matched elements or set the HTML contents of every matched element.

$("#this_p5").html("<a href='#'><span class='red'>more ...</span></a>");

.text()

Get the combined text contents of each element in the set of matched elements, including their descendants, or set the text contents of the matched elements.

$("#this_div p:last-child").text("Some new text.");

.val()

Get the current value of the first element in the set of matched elements or set the value of every matched element.

$("#this_div2 p:last-child").text($("#this_div2 input").val());

.toggle(duration ] [, complete ])

Display or hide the matched elements.

$("#this_p6").toggle(2000, function() { $("#this_p6b").text("All done!"); });

.hide([duration ] [, complete ])

Hide the matched elements.

$("#this_p7").hide(2000, function() { $("#this_p7b").text("All done!"); });

.show([duration ] [, complete ])

.addClass adds the specified class to each element in the set of matched elements.

$("#this_p7").show(2000, function() { $("#this_p8").show(2000, function() { $("#this_p8b").text("All done!"); }); });

.slideDown([duration ] [, complete ])

Display the matched elements with a sliding motion.

$("#this_p9").slideDown(3000, function() { $("#this_p9b").slideDown(3000, function() { $("#this_p9b").text("All done!"); }); });

.slideToggle([duration ] [, easing ] [, complete ])

Display or hide the matched elements with a sliding motion.

$("#this_p10").slideToggle(3000, function() { $("#this_p10b").slideToggle("slow", function() { $("#this_p10b").text("All done!"); }); });

.slideUp([duration ] [, complete ])

Hide the matched elements with a sliding motion.

$("#this_p11").slideUp(3000, function() { $("#this_p11b").slideUp("slow", function() { alert("All done!"); }); });

.fadeOut([duration ] [, complete ])

Hide the matched elements by fading them to transparent.

Hide the matched elements by fading them to transparent.

$("#this_p12").fadeOut(3000, function() { $("#this_p12b").fadeOut(5000); });

.fadeIn([duration ] [, complete ])

Display the matched elements by fading them to opaque.

$("#this_p13").fadeIn(3000, function() { $("#this_p13b").text("All done!"); });

.focus(handler)

Bind an event handler to the “focus” JavaScript event, or trigger that event on an element.

$("#focus input").focus(function() { $("#this_p14b").text("You clicked inside the box!"); });

.click(handler)

Bind an event handler to the "click" JavaScript event, or trigger that event on an element.

$("#click").click(function() { $("#this_p15b").text("You clicked the button!"); });
